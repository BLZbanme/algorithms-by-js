## 标准二分查找

```javascript
function binarySearch1(nums, target) {
    let lo = 0;
    let hi = nums.length - 1;
    while (lo <= hi) {
        let mid = lo + ((hi - lo) >> 1);
        if (nums[mid] < target) {
            lo = mid + 1;
        }
        else if (nums[mid] > target) {
            hi = mid - 1;
        }
        return mid;
    }
    return lo;
}
```
